115  private void InitGradients()
116  {
117  for (int i = 0; i < GradientSizeTable; i++)
118  {
119  double z = 1f - 2f * this._random.NextDouble();
120  double r = Math.Sqrt(1f - z * z);
121  double theta = 2 * Math.PI * this._random.NextDouble();
122  this._gradients[i * 3] = r * Math.Cos(theta);
123  this._gradients[i * 3 + 1] = r * Math.Sin(theta);
124  this._gradients[i * 3 + 2] = z;
